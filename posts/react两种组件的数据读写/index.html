React两种组件的数据读写

2020年3月11日

今天简单分享一下React中组件的数据读写

类组件

//定义一个类组件

class Component extends React.Component {
    render() {
        return (
            <div>类组件</div>
        );
    }
}

//使用类组件

<Component/>

类组件中使用外部数据props

//父组件给我传了一个data

class Component extends React.Component {
    constructor(props){   //没有数据时可以省略
        super(props)
    }
    render() {
        return (
            <div>
            {this.props.data}    
            </div>
        );
    }
}

类组件中读写数据

class Component extends React.Component {
    constructor(props){   
        super(props)
        this.state = {   //初始化数据
            data : 0
        }
    }
    add = ()=>{
        this.setState({data : this.state.data + 1})  
    }
    render() {
        return (
            <div>
                <button onClick={this.add}> +1 </button>
            </div>
        );
    }
}

!!!! 类组件中修改数据要使用setState

函数组件

//定义一个函数组件

function Component(){
    return (
        <div>函数组件</div>
    )
}

//使用

<Component/>

函数组件中使用外部数据props

//父组件传了一个data

function Component(props){
    return (
        <div>
            {props.data}
        </div>
    )
}

函数组件中读写数据

function Component(){
    const [n,setN] = React.useState(0)(0)
    return (
        <div>
           {n}
           <button onClick={()=>setN(n+1)}> +1 </button>
        </div>
    )
}

const [n,setN] = React.useState(0)(0)

  • n 用来读取数据
  • setN 用来修改数据
  • React.useState(0) 初始化数据

React.useState(0) 可以先引入useState，这样就能直接使用useState

import {useState} from 'react'
